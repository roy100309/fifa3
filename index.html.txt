<script>
    // 상태 초기 설정 값 정의
    let hasPc = true;
    let hasTop = true;
    let couponRate = 0;

    // 공통 옵션 선택 제어 스크립트
    function toggleBenefit(type) {
        if (type === 'pc') {
            hasPc = !hasPc;
            document.getElementById('btn-pc').classList.toggle('active', hasPc);
        } else if (type === 'top') {
            hasTop = !hasTop;
            document.getElementById('btn-top').classList.toggle('active', hasTop);
        }
        calculate();
    }

    // 쿠폰 수치 제한 연산 시스템
    function adjustCoupon(amount) {
        couponRate += amount;
        if (couponRate < 0) couponRate = 0;
        if (couponRate > 50) couponRate = 50; 

        const display = document.getElementById('coupon-display');
        if (couponRate === 0) {
            display.innerText = "쿠폰 미적용 (0%)";
        } else {
            display.innerText = `${couponRate}% 할인 쿠폰`;
        }
        calculate();
    }

    // 한글 가독성 단위 포맷팅 모듈
    function formatKorean(num) {
        if (num === 0) return "0 BP";
        let jo = Math.floor(num / 1000000000000);
        let eok = Math.floor((num % 1000000000000) / 100000000);
        
        let result = [];
        if (jo > 0) result.push(`${jo}조`);
        if (eok > 0) result.push(`${eok}억`);
        return result.join(' ') + " BP";
    }

    // 실시간 연산 핵심 비즈니스 로직
    function calculate() {
        const inputVal = parseFloat(document.getElementById('price-input').value);
        const hint = document.getElementById('live-hint');
        
        if (isNaN(inputVal) || inputVal <= 0) {
            hint.innerText = "";
            document.getElementById('res-receive').innerText = "0 BP";
            return;
        }

        // 1조 기준 수치 변환 처리
        const realBP = Math.round(inputVal * 1000000000000);
        hint.innerText = `인식된 금액: ${realBP.toLocaleString()} BP`;

        // 할인율 산출
        const totalDiscount = (hasPc ? 30 : 0) + (hasTop ? 20 : 0) + couponRate;
        document.getElementById('res-discount').innerText = `${totalDiscount}%`;

        // 수수료율 및 인게임 정수 반올림 동기화 반영
        const actualFeeRate = 40 * (1 - (totalDiscount / 100));
        document.getElementById('res-fee-rate').innerText = `${actualFeeRate.toFixed(1).replace('.0', '')}%`;

        const receiveAmount = Math.round(realBP * (1 - (actualFeeRate / 100)));
        document.getElementById('res-receive').innerText = `${receiveAmount.toLocaleString()} BP (${formatKorean(receiveAmount)})`;
    }

    // 문서 시작 시 즉각 기본 계산 대기 상태 활성화
    calculate();
</script>
